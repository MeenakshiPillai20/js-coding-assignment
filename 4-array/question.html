Qu 4. Write a program to print sum of all even number and odd number in an array?

    const findEvenOdd = (n, arr) => 
    {
        let sum=[];
        let even=0,odd=0;
        for (let i=0;i< n;i++){
          if (arr[i]%2===0)
            even+=arr[i];
          else
            odd+=arr[i];
          
        }
        sum[0]=even;
        sum[1]=odd;
        return sum;
    };

          if(val>arr[i]){
            val = arr[i];
          }
        }
        return val;
